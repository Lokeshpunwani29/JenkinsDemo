pipeline {
    agent any

    tools {
        jdk 'Java17'
        maven 'Maven3'
    }

    stages {

        stage('Verify Tools') {
            steps {
                bat 'java -version'
                bat 'mvn -version'
            }
        }

        stage('Build') {
            steps {
                bat 'mvn clean package'
            }
        }

        stage('Run JAR') {
            steps {
                bat 'java -jar target\\JenkinsDemo-0.0.1-SNAPSHOT.jar'
            }
        }
    }
}
